const data = db.jrfsrves.aggregate([
    {
        $lookup: {
            from: "users",
            localField: "userId",
            foreignField: "_id",
            as: "user"
        }
    },
    {
        $addFields: {
            userId: { $arrayElemAt: ["$user", 0] }
        }
    },
    {
        $project: {
            user: 0 // Optionally remove the 'user' field from the result
        }
    }
]);

data.forEach(function(doc) {
    if(doc.userId){
        db.jrfsrves.updateOne(
            { "_id": doc._id }, 
            { 
                "$set": {
                    "guideName": doc.userId.name,
                    "schoolName": doc.userId.department
                } 
            }
        );
    }
});
